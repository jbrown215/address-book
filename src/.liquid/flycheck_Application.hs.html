 52: -- | This function allocates resources (such as a database connection pool),
 53: -- performs initialization and returns a foundation datatype value. This is also
 54: -- the place to put your migrate statements to have automatic database
 55: -- migrations handled by Yesod.
 56: makeFoundation :: AppSettings -> IO App
 57: makeFoundation appSettings = do
 58:     -- Some basic initializations: HTTP connection manager, logger, and static
 59:     -- subsite.
 60:     appHttpManager <- newManager
 61:     appLogger <- newStdoutLoggerSet defaultBufSize >>= makeYesodLogger
 62:     appStatic <-
 63:         (if appMutableStatic appSettings then staticDevel else static)
 64:         (appStaticDir appSettings)
 65: 
 66:     -- We need a log function to create a connection pool. We need a connection
 67:     -- pool to create our foundation. And we need our foundation to get a
 68:     -- logging function. To get out of this loop, we initially create a
 69:     -- temporary foundation without a real connection pool, get a log function
 70:     -- from there, and then create the real foundation.
 71:     let mkFoundation appConnPool = App {..}
 72:         -- The App {..} syntax is an example of record wild cards. For more
 73:         -- information, see:
 74:         -- https://ocharles.org.uk/blog/posts/2014-12-04-record-wildcards.html
 75:         tempFoundation = mkFoundation $ error "connPool forced in tempFoundation"
 76:         logFunc = messageLoggerSource tempFoundation appLogger
 77: 
 78:     -- Create the database connection pool
 79:     pool <- flip runLoggingT logFunc $ createSqlitePool
 80:         (sqlDatabase $ appDatabaseConf appSettings)
 81:         (sqlPoolSize $ appDatabaseConf appSettings)
 82: 
 83:     -- Perform database migration using our application's logging settings.
 84:     runLoggingT (runSqlPool (runMigration migrateAll) pool) logFunc
 85: 
 86:     -- Return the foundation
 87:     return $ mkFoundation pool

 89: -- | Convert our foundation to a WAI Application by calling @toWaiAppPlain@ and
 90: -- applying some additional middlewares.
 91: makeApplication :: App -> IO Application
 92: makeApplication foundation = do
 93:     logWare <- makeLogWare foundation
 94:     -- Create the WAI application and apply middlewares
 95:     appPlain <- toWaiAppPlain foundation
 96:     return $ logWare $ defaultMiddlewaresNoLogging appPlain
 97: 
 98: makeLogWare :: App -> IO Middleware
 99: makeLogWare foundation =
100:     mkRequestLogger def
101:         { outputFormat =
102:             if appDetailedRequestLogging $ appSettings foundation
103:                 then Detailed True
104:                 else Apache
105:                         (if appIpFromHeader $ appSettings foundation
106:                             then FromFallback
107:                             else FromSocket)
108:         , destination = Logger $ loggerSet $ appLogger foundation
109:         }

112: -- | Warp settings for the given foundation value.
113: warpSettings :: App -> Settings
114: warpSettings foundation =
115:       setPort (appPort $ appSettings foundation)
116:     $ setHost (appHost $ appSettings foundation)
117:     $ setOnException (\_req e ->
118:         when (defaultShouldDisplayException e) $ messageLoggerSource
119:             foundation
120:             (appLogger foundation)
121:             $(qLocation >>= liftLoc)
122:             "yesod"
123:             LevelError
124:             (toLogStr $ "Exception from Warp: " ++ show e))
125:       defaultSettings
